139  std::true_type, std::true_type)
140  // Only remove neutral fixations.
141  // Record fixations.
142  // The recording is in order of mutation position.
143  {
144  if (mutations[i].neutral)
145  {
146  // Record the fixation here,
147  // as we will recycle this variant
148  auto loc = std::lower_bound(
149  fixations.begin(), fixations.end(),
150  mutations[i].pos,
151  [](const typename mcont_t::value_type &m,
152  const double val) { return m.pos < val; });
153  auto d = std::distance(fixations.begin(), loc);
154  fixations.insert(loc, mutations[i]);
155  fixation_times.insert(fixation_times.begin() + d,
156  generation);
157  process_mutation_index(mutations, mut_lookup, i);
158  mcounts[i] = 0;
159  }
160  else
161  {
162  // The logic here is tougher. We are preserving
163  // selected mutations, and thus we only
164  // want to record their fixations once. Thus,
165  // we have to make sure that this fixation
166  // hasn't been recorded at all.
167  auto loc_range = std::equal_range(
168  fixations.begin(), fixations.end(),

170  detail::equal_range_comparison());

172  mutations[i])
173  == loc_range.second)
174  {
175  auto d = std::distance(fixations.begin(),
176  loc_range.first);
177  fixations.insert(loc_range.first,

193  const uint_t twoN, const uint_t generation,
194  const preserve_selected_fixations preserve,
195  const record_fixations record)
196  /*! Mark mutations for recycling.
197  *
198  * This function should be called immediately after simplification and mutation counting.
199  *
200  * \param pop A mutation container
201  * \param mcounts_from_preserved_nodes A container recording the counts of each mutation in ancient samples
202  * \param twoN Twice the current population size
203  * \param generation Current generation/time step in the simulation
204  * \param preserve Policy for handling selected fixations. See below.
205  * \param record Policy for recording fixation events. See below.
206  *
207  * A mutation is marked for recycling if one of the following conditions holds:
208  * 1. The sum of \a pop.mcounts and \a mcounts_from_preserved_nodes is zero.
209  * 2. \a pop.mcounts == \a twoN, mcounts_from_preserved_nodes is zero,
210  * and \a record is std::false_type or the mutation is neutral.
211  *
212  * Condition 1 refers to extinct mutations and condition 2 refers to fixations.
213  *
214  * When \a preserve is std::true_type, selected fixations are retained
215  * in the population. We do this because simulations of phenotypes (as opposed
216  * to relative fitness) require tracking the contribution of fixation to
217  * trait values.
218  *
219  * All variants matching the above criteria have their record in \a pop.mcounts
220  * set to zero and their position is removed from \a pop.mut_lookup. Further, the
221  * mutation's position in \a pop.mutations is set to std::numeric_limits<double>::max(),
222  * to signify an "invalid" mutation.
223  *
224  * If \a record is std::false_type, no fixation recording takes place. However, if
225  * \a record is std::true_type, a record is entered into \a pop.fixations and
226  * \a pop.fixation_times. If \a preserve is also std::true_type, the recording is
227  * slightly more expensive because we have to guard against repeated recording. For this
228  * case, the fixations and fixation times containers in \a pop are kept sorted by fixation position.
229  *
230  * An advanced use of this function is to create branchless code with respect to
231  * fixation handling. One can generate a closure wrapping this function with the desired
232  * values of \a preserve and \a record at the start of a simulation, thus avoiding a bunch of
233  * "if" statements prior to each call.
234  *
235  * \note There are several caveats to fixation recording (\a record == std::true_type). First,
236  * it is only as accurate as the time steps between simplifications is small. \a pop will only
237  * contain exact fixation times if simplification occurs every generation. The second limitation
238  * is that it will not give correct results if ancient samples are being tracked in a simulation.
239  * This has to do with a "todo" item for this function described below. Thus, fixation recording
240  * with this function should be considered limited to the case of regular simplification and no
241  * ancient sample preservation.
242  *
243  * \warning The \a preserve argument needs to be consistent with the desired behavior of
244  * ts::remove_fixations_from_gametes.
245  *
246  * \version 0.7.0 Added to library
247  * \version 0.7.1 Updated to change recycled mutation positions to max value of a double.
248  * Refactor API to take compile-time policies for fixation handling.
249  *
250  * \todo Improve treatment of fixations by allowing for variants fixed in alive AND
251  * ancient samples to be flagged.
252  * \todo Return a recycling queue?
253  */
254  {
255  for (std::size_t i = 0; i < pop.mcounts.size(); ++i)
256  {
257  if (mcounts_from_preserved_nodes[i] == 0)
258  {
259  if (pop.mcounts[i] > twoN)
260  {
261  throw std::runtime_error(
262  "mutation count out of range");
263  }
264  if (pop.mcounts[i] == twoN)
265  {
266  detail::process_fixations(
267  pop.mutations, pop.mcounts,
268  pop.fixations, pop.fixation_times,
269  pop.mut_lookup, generation, i,
270  preserve, record);
271  }
272  else if (pop.mcounts[i] == 0)
273  {
274  detail::process_mutation_index(
275  pop.mutations, pop.mut_lookup, i);
276  }
277  }
278  }
