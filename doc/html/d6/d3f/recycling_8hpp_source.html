43  make_mut_queue(
44  const std::vector<std::size_t> &preserved_mutation_indexes,
45  const std::size_t num_mutations)
46  /// \brief Make a mutation recycling queue for simulations with tree sequences
47  /// \param preserved_mutation_indexes Vector of preserved mutation indexes returned by simplification
48  /// \param num_mutations The total number of mutations currently allocated in the population
49  ///
50  /// \returns flagged_mutation_queue
51  ///
52  /// \version 0.7.3 Added to fwdpp
53  ///
54  /// This overload may be preferable to the other when the following conditions apply:
55  /// 1. Mutation counts in the entire simulation are not of interest during the simulation.
56  /// 2. There is no need/wish to remove fixations from the gametes/tables during the simulation.
57  /// 3. There are large numbers of ancient samples being recorded.
58  ///
59  /// The first two conditions are required for correct results. The third condition is optional,
60  /// but big speedups will be seen for that case.
61  ///
62  /// This function generates a recycling queue using a fast O(N) algorithm.
63  /// This, it will outperform the other overload based
64  /// on tree traversal when the number of trees is very large, as is the case when large numbers
65  /// of ancestral samples are registered during a simulation.
66  ///
67  {
68  std::vector<std::size_t> mindexes(num_mutations);
69  std::iota(mindexes.begin(), mindexes.end(), 0);
70  for (auto i : preserved_mutation_indexes)
71  {
72  mindexes[i] = std::numeric_limits<std::size_t>::max();
73  }
74  flagged_mutation_queue::value_type rv;
75  for (auto i : mindexes)
76  {
77  if (i != std::numeric_limits<std::size_t>::max())
78  {
79  rv.push(i);
80  }
81  }
82  return flagged_mutation_queue(std::move(rv));
83  }

179  typename lookup_table>
180  inline void
181  process_fixations(mcont_t &mutations,
182  mutation_count_container &mcounts,
183  mcont_t &fixations,
184  std::vector<uint_t> &fixation_times,
185  lookup_table &mut_lookup,
186  const uint_t generation, const std::size_t i,
187  std::true_type, std::true_type)
188  // Only remove neutral fixations.
189  // Record fixations.
190  // The recording is in order of mutation position.
191  {
192  if (mutations[i].neutral)
193  {
194  // Record the fixation here,
195  // as we will recycle this variant
196  auto loc = std::lower_bound(
197  fixations.begin(), fixations.end(),
198  mutations[i].pos,
199  [](const typename mcont_t::value_type &m,
200  const double val) { return m.pos < val; });
201  auto d = std::distance(fixations.begin(), loc);
202  fixations.insert(loc, mutations[i]);
203  fixation_times.insert(fixation_times.begin() + d,
204  generation);
205  process_mutation_index(mutations, mut_lookup, i);
206  mcounts[i] = 0;
207  }
208  else
209  {
210  // The logic here is tougher. We are preserving
211  // selected mutations, and thus we only
212  // want to record their fixations once. Thus,
213  // we have to make sure that this fixation
214  // hasn't been recorded at all.
215  auto loc_range = std::equal_range(
216  fixations.begin(), fixations.end(),
217  mutations[i].pos,
218  detail::equal_range_comparison());
219  if (std::find(loc_range.first, loc_range.second,
220  mutations[i])
221  == loc_range.second)
222  {
223  auto d = std::distance(fixations.begin(),
224  loc_range.first);
225  fixations.insert(loc_range.first,
226  mutations[i]);
227  fixation_times.insert(
228  fixation_times.begin() + d, generation);
229  }

244  /*! Mark mutations for recycling.
245  *
246  * This function should be called immediately after simplification and mutation counting.
247  *
248  * \param pop A mutation container
249  * \param mcounts_from_preserved_nodes A container recording the counts of each mutation in ancient samples
250  * \param twoN Twice the current population size
251  * \param generation Current generation/time step in the simulation
252  * \param preserve Policy for handling selected fixations. See below.
253  * \param record Policy for recording fixation events. See below.
254  *
255  * A mutation is marked for recycling if one of the following conditions holds:
256  * 1. The sum of \a pop.mcounts and \a mcounts_from_preserved_nodes is zero.
257  * 2. \a pop.mcounts == \a twoN, mcounts_from_preserved_nodes is zero,
258  * and \a record is std::false_type or the mutation is neutral.
259  *
260  * Condition 1 refers to extinct mutations and condition 2 refers to fixations.
261  *
262  * When \a preserve is std::true_type, selected fixations are retained
263  * in the population. We do this because simulations of phenotypes (as opposed
264  * to relative fitness) require tracking the contribution of fixation to
265  * trait values.
266  *
267  * All variants matching the above criteria have their record in \a pop.mcounts
268  * set to zero and their position is removed from \a pop.mut_lookup. Further, the
269  * mutation's position in \a pop.mutations is set to std::numeric_limits<double>::max(),
270  * to signify an "invalid" mutation.
271  *
272  * If \a record is std::false_type, no fixation recording takes place. However, if
273  * \a record is std::true_type, a record is entered into \a pop.fixations and
274  * \a pop.fixation_times. If \a preserve is also std::true_type, the recording is
275  * slightly more expensive because we have to guard against repeated recording. For this
276  * case, the fixations and fixation times containers in \a pop are kept sorted by fixation position.
277  *
278  * An advanced use of this function is to create branchless code with respect to
279  * fixation handling. One can generate a closure wrapping this function with the desired
280  * values of \a preserve and \a record at the start of a simulation, thus avoiding a bunch of
281  * "if" statements prior to each call.
282  *
283  * \note There are several caveats to fixation recording (\a record == std::true_type). First,
284  * it is only as accurate as the time steps between simplifications is small. \a pop will only
285  * contain exact fixation times if simplification occurs every generation. The second limitation
286  * is that it will not give correct results if ancient samples are being tracked in a simulation.
287  * This has to do with a "todo" item for this function described below. Thus, fixation recording
288  * with this function should be considered limited to the case of regular simplification and no
289  * ancient sample preservation.
290  *
291  * \warning The \a preserve argument needs to be consistent with the desired behavior of
292  * ts::remove_fixations_from_gametes.
293  *
294  * \version 0.7.0 Added to library
295  * \version 0.7.1 Updated to change recycled mutation positions to max value of a double.
296  * Refactor API to take compile-time policies for fixation handling.
297  *
298  * \todo Improve treatment of fixations by allowing for variants fixed in alive AND
299  * ancient samples to be flagged.
300  * \todo Return a recycling queue?
301  */
302  {
303  for (std::size_t i = 0; i < pop.mcounts.size(); ++i)
304  {
305  if (mcounts_from_preserved_nodes[i] == 0)
306  {
307  if (pop.mcounts[i] > twoN)
308  {
309  throw std::runtime_error(
310  "mutation count out of range");
311  }
312  if (pop.mcounts[i] == twoN)
313  {
314  detail::process_fixations(
315  pop.mutations, pop.mcounts,
316  pop.fixations, pop.fixation_times,
317  pop.mut_lookup, generation, i,
318  preserve, record);
319  }
320  else if (pop.mcounts[i] == 0)
321  {
322  detail::process_mutation_index(
323  pop.mutations, pop.mut_lookup, i);
324  }
325  }
326  }
