102 std::size_t
103 infsites_TwoDmutation_additive(queue_t &recycling_bin, mcont_t &mutations,
104  const gsl_rng *r, lookup_table_t &lookup,
105  const fwdpp::uint_t &generation,
106  const double pselected,
107  const position_function &posmaker,
108  const double sigma_x, const double sigma_y,
109  const double rho,
110  const decltype(TwoDmutation::xtra) x = 0)
111 {
112  TwoDmutation::array_t s{ 0., 0. }, h{ 1., 1. };
113 
114  double pos = posmaker();
115  while (lookup.find(pos) != lookup.end())
116  {
117  pos = posmaker();
118  }
119 
120  if (gsl_rng_uniform(r) < pselected)
121  {
122  gsl_ran_bivariate_gaussian(r, sigma_x, sigma_y, rho, &s[0], &s[1]);
123  }
124 
125  auto idx = fwdpp::fwdpp_internal::recycle_mutation_helper(
126  recycling_bin, mutations, std::move(s), std::move(h), pos, generation,
127  x);
128  lookup.emplace(pos, idx);
129  return idx;
130 }
