102 std::size_t
103 infsites_TwoDmutation_additive(
104  fwdpp::flagged_mutation_queue &recycling_bin, mcont_t &mutations,
105  const gsl_rng *r, lookup_table_t &lookup, const fwdpp::uint_t &generation,
106  const double pselected, const position_function &posmaker,
107  const double sigma_x, const double sigma_y, const double rho,
108  const decltype(TwoDmutation::xtra) x = 0)
109 {
110  TwoDmutation::array_t s{ 0., 0. }, h{ 1., 1. };
111 
112  double pos = posmaker();
113  while (lookup.find(pos) != lookup.end())
114  {
115  pos = posmaker();
116  }
117 
118  if (gsl_rng_uniform(r) < pselected)
119  {
120  gsl_ran_bivariate_gaussian(r, sigma_x, sigma_y, rho, &s[0], &s[1]);
121  }
122 
123  auto idx = fwdpp::recycle_mutation_helper(recycling_bin, mutations,
124  std::move(s), std::move(h), pos,
125  generation, x);
126  lookup.emplace(pos, idx);
127  return idx;
128 }
