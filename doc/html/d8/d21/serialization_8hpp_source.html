202  /*! \brief Write a fwdpp::ts::table_collection to a binary format.
203  *
204  * \param o A model of std::ostream
205  * \param tables A fwdpp::ts::table_collection
206  *
207  * The result is that the data in \a tables are written
208  * in a machine-readable format to \a o.
209  *
210  * This function always outputs to the format version
211  * specified by fwdpp::ts::io::TS_TABLES_VERSION.
212  *
213  * \version 0.7.4 Tables are written in a single call
214  */
215  {
216  o << "fwdppts";
217  fwdpp::io::scalar_writer sw;
218  sw(o, &TS_TABLES_VERSION);
219  auto L = tables.genome_length();
220  sw(o, &L);
221  sw(o, &tables.edge_offset);
222  std::size_t num_edges = tables.edge_table.size(),
223  num_nodes = tables.num_nodes(),
224  num_mutations = tables.mutation_table.size(),
225  num_sites = tables.site_table.size();
226  sw(o, &num_edges);
227  sw(o, &num_nodes);
228  sw(o, &num_mutations);
229  sw(o, &num_sites);
230  if (!tables.edge_table.empty())
231  {
232  o.write(reinterpret_cast<const char*>(
233  tables.edge_table.data()),
234  tables.edge_table.size() * sizeof(edge));
235  }
236  if (!tables.node_table.empty())
237  {
238  o.write(reinterpret_cast<const char*>(
239  tables.node_table.data()),
240  tables.node_table.size() * sizeof(node));
241  }
242  if (!tables.mutation_table.empty())
243  {
244  o.write(reinterpret_cast<const char*>(
245  tables.mutation_table.data()),
246  tables.mutation_table.size()
247  * sizeof(mutation_record));
248  }
249  if (!tables.site_table.empty())
250  {
251  o.write(reinterpret_cast<const char*>(
252  tables.site_table.data()),
253  tables.site_table.size() * sizeof(site));
254  }
255  std::size_t num_preserved_samples
256  = tables.preserved_nodes.size();
257  sw(o, &num_preserved_samples);
258  if (num_preserved_samples)
259  {
260  sw(o, tables.preserved_nodes.data(),
261  num_preserved_samples);
