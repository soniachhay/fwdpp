1 /* \include juvenile_migration.cc
2  *
3  * Juvenile migration.
4  * Selection.
5  * Constant pop size.
6  * Unequal migration rates.
7  *
8  * The goal is to show how to use a population
9  * object to handle details of population structure.
10  *
11  * The outline of the scheme is as follows:
12  *
13  * 1. Offspring are generated in sorted order by deme
14  * label. In other words, deme 1 before deme 2.
15  *
16  * 2. Each generation, we migrate first. Thus,
17  * we may start a generation with parents like this:
18  *
19  * Parent Deme
20  * 0 0
21  * 1 0
22  * 2 0
23  * 3 1
24  * 4 1
25  * 5 1
26  *
27  * After migration, we may end up with:
28  *
29  * Parent Deme
30  * 0 1*
31  * 1 0
32  * 2 0
33  * 3 1
34  * 4 0*
35  * 5 1
36  *
37  * An asterisk represent migrant individuals.
38  *
39  * We will generate efficient lookup tables to sample individuals
40  * proportional to their within-deme fitness, post-migration.
41  *
42  * These lookup tables will map to the indexes of the parents of each
43  * deme:
44  *
45  * Lookup 1:
46  *
47  * Index Parent
48  * 0 1
49  * 1 2
50  * 2 4
51  *
52  * Lookup 2:
53  * Index Parent
54  * 0 0
55  * 1 3
56  * 1 5
57  *
58  * The code is not safe for real-world use. It doesn't handle the corner
59  * case of all of deme 1 migrating into deme 2, leaving deme 1 "extinct"
60  * in the next generation. Such edge cases are clearly important, but
61  * the goal here it to show the book-keeping of parental fitnesses
62  * and the mapping back to deme labels.
63  */

83 struct parent_lookup_tables
84 // Our object for choosing parents each generation
85 {
86  // These return indexes of parents from demes 1 and 2,
87  // resp, chosen in O(1) time proportional to
88  // relative fitness within each deme
89  fwdpp::gsl_ran_discrete_t_ptr lookup1, lookup2;
90  // These vectors map indexes returned from sampling
91  // lookup1 and lookup2 to diploids in the population
92  // object.
93  std::vector<std::size_t> parents1, parents2;
94 };

96 template <typename fitness_fxn>
97 parent_lookup_tables
98 migrate_and_calc_fitness(const gsl_rng *r, diploid_population &pop,
99  const fitness_fxn &wfxn, const fwdpp::uint_t N1,
100  const fwdpp::uint_t N2, const double m12,
101  const double m21)
102 // This function will be called at the start of each generation.
103 // The main goal is to return the lookup tables described above.
104 // But, "while we're at it", it does some other stuff that
105 // needs to be done at the start of each generation.
106 // Neither the most rigorous nor the most efficient:
107 // 1. Ignores probability of back-migration.
108 // 2. Allocates 4 vectors each generation.
109 {
110  parent_lookup_tables rv;
111 
112  // Temp containers for fitnesses in each deme,
113  // post-migration
114  std::vector<double> w1, w2;
115 
116  // Pick no. migrants 1 -> 2 and 2 -> 1.
117  unsigned nmig12 = gsl_ran_poisson(r, static_cast<double>(N1) * m12);
118  unsigned nmig21 = gsl_ran_poisson(r, static_cast<double>(N2) * m21);
119 
120  // Fill a vector of N1 zeros and N2 ones:
121  std::vector<fwdpp::uint_t> deme_labels(N1, 0);
122  deme_labels.resize(N1 + N2, 1);
123  assert(deme_labels.size() == pop.diploids.size());
124 
125  // Set up source and destination containers
126  // for sampling w/o replacement
127  std::vector<std::size_t> individuals(N1 + N2);
128  std::iota(std::begin(individuals), std::end(individuals), 0);
129  std::vector<std::size_t> migrants(std::max(nmig12, nmig21));
130 
131  // Who is migrating 1 -> 2?
132  gsl_ran_choose(r, migrants.data(), nmig12, individuals.data(), N1,
133  sizeof(std::size_t));
134  for (std::size_t i = 0; i < nmig12; ++i)
135  {
136  deme_labels[migrants[i]] = !deme_labels[migrants[i]];
137  }
138 
139  // Exact same logic for migrants 2 -> 1
140  gsl_ran_choose(r, migrants.data(), nmig21, individuals.data() + N1, N2,
141  sizeof(std::size_t));
142  for (std::size_t i = 0; i < nmig21; ++i)
143  {
144  deme_labels[migrants[i]] = !deme_labels[migrants[i]];
145  }
146 
147  // Go over all parents, set haploid_genomes counts to zero,
148  // and put individual IDs and fitnesses into
149  // the right vectors:
150  for (std::size_t i = 0; i < deme_labels.size(); ++i)
151  {
152  // fwdpp requires that we zero out haploid_genome
153  // counts each generation. Since we're looping
154  // over diploids here, now is a good time to
155  // handle this task, which saves us from having to
156  // do another O(N1+N2) loop:
157  pop.haploid_genomes[pop.diploids[i].first].n
158  = pop.haploid_genomes[pop.diploids[i].second].n = 0;
159  if (deme_labels[i] == 0)
160  {
161  rv.parents1.push_back(i);
162  w1.push_back(
163  wfxn(pop.diploids[i], pop.haploid_genomes, pop.mutations));
164  }
165  else
166  {
167  rv.parents2.push_back(i);
168  w2.push_back(
169  wfxn(pop.diploids[i], pop.haploid_genomes, pop.mutations));
170  }
171  }
172 
173  // Set up our lookup tables:
174  rv.lookup1.reset(gsl_ran_discrete_preproc(rv.parents1.size(), w1.data()));
175  rv.lookup2.reset(gsl_ran_discrete_preproc(rv.parents2.size(), w2.data()));
176  return rv;
177 };

180 void
181 evolve_two_demes(const gsl_rng *r, diploid_population &pop,
182  const fwdpp::uint_t N1, const fwdpp::uint_t N2,
183  const double m12, const double m21, const double mu,
184  const fitness_fxn &wfxn, const rec_fxn &recfxn,
185  const mut_fxn &mutfxn)
186 {
187  // Handle mutation/haploid_genome "recycling":
188  auto mut_recycling_bin = fwdpp::make_mut_queue(pop.mcounts);
189  auto gam_recycling_bin = fwdpp::make_haploid_genome_queue(pop.haploid_genomes);
190 
191  // Migration and build lookup tables:
192  auto lookups = migrate_and_calc_fitness(r, pop, wfxn, N1, N2, m12, m21);
193 
194 #ifndef NDEBUG
195  for (const auto &g : pop.haploid_genomes)
196  assert(!g.n);
197 #endif
198 
199  // Copy parents
200  const auto parents(pop.diploids);
201 
202  // Fill in the next generation!
203  // We generate the offspring for deme 1 first,
204  // and then for deme 2
205  for (fwdpp::uint_t i = 0; i < N1 + N2; ++i)
206  {
207  std::size_t p1 = std::numeric_limits<std::size_t>::max();
208  std::size_t p2 = std::numeric_limits<std::size_t>::max();
209  if (i < N1) // pick parents from pop 1
210  {
211  p1 = lookups.parents1[gsl_ran_discrete(
212  r, lookups.lookup1.get())];
213  p2 = lookups.parents1[gsl_ran_discrete(
214  r, lookups.lookup1.get())];
215  }
216  else // pick parents from pop 2
217  {
218  p1 = lookups.parents2[gsl_ran_discrete(
219  r, lookups.lookup2.get())];
220  p2 = lookups.parents2[gsl_ran_discrete(
221  r, lookups.lookup2.get())];
222  }
223  assert(p1 < parents.size());
224  assert(p2 < parents.size());
225 
226  /*
227  These are the haploid_genomes from each parent.
228  */
229  auto p1g1 = parents[p1].first;
230  auto p1g2 = parents[p1].second;
231  auto p2g1 = parents[p2].first;
232  auto p2g2 = parents[p2].second;
233 
234  // "Mendel"
235  if (gsl_rng_uniform(r) < 0.5)
236  std::swap(p1g1, p1g2);
237  if (gsl_rng_uniform(r) < 0.5)
238  std::swap(p2g1, p2g2);
239 
240  mutate_recombine_update(r, pop.haploid_genomes, pop.mutations,
241  std::make_tuple(p1g1, p1g2, p2g1, p2g2),
242  recfxn, mutfxn, mu, gam_recycling_bin,
243  mut_recycling_bin, pop.diploids[i],
244  pop.neutral, pop.selected);
245  }
246  fwdpp::debug::validate_sum_haploid_genome_counts(pop.haploid_genomes,
247  2 * pop.diploids.size());
248 #ifndef NDEBUG
249  for (const auto &dip : pop.diploids)
250  {
251  assert(pop.haploid_genomes[dip.first].n > 0);
252  assert(pop.haploid_genomes[dip.first].n <= 2 * (N1 + N2));
253  assert(pop.haploid_genomes[dip.second].n > 0);
254  assert(pop.haploid_genomes[dip.second].n <= 2 * (N1 + N2));
255  }
256 #endif
257 
258  // Update mutation counts
259  fwdpp::fwdpp_internal::process_haploid_genomes(pop.haploid_genomes, pop.mutations,
260  pop.mcounts);
261 
262  assert(pop.mcounts.size() == pop.mutations.size());
263 #ifndef NDEBUG
264  for (const auto &mc : pop.mcounts)
265  {
266  assert(mc <= 2 * (N1 + N2));
267  }
268 #endif
269  fwdpp::debug::validate_pop_data(pop);
270 
271  // Prune fixations from haploid_genomes
272  fwdpp::fwdpp_internal::haploid_genome_cleaner(pop.haploid_genomes, pop.mutations,
273  pop.mcounts, 2 * (N1 + N2),
274  std::true_type());
275 }
