204  const std::vector<fwdpp::ts::TS_NODE_INT> &sample_list)
205 {
206  fwdpp::ts::tree_visitor mti(tables, sample_list);
207  while (mti(std::true_type(), std::true_type()))
208  {
209  auto &tree = mti.tree();
210  for (auto i : sample_list)
211  {
212  auto p = i;
213  while (p != -1)
214  {
215  auto l = tree.left_sample[p];
216  auto ogl = l;
217  if (l != -1)
218  {
219  auto r = tree.right_sample[p];
220  int ns = 0;
221  while (true)
222  {
223  ++ns;
224  if (l == r)
225  {
226  break;
227  }
228  l = tree.next_sample[l];
229  if (l == ogl)
230  {
231  throw std::runtime_error(
232  "loopback "

234  }
235  }
236  if (ns != tree.leaf_counts[p])
237  {
238  throw std::runtime_error(
239  "bad sample interval");
240  }
241  }
242  p = tree.parents[p];
243  }
