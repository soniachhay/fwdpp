73  void
74  split_breakpoints(
75  const std::vector<double>& breakpoints,
76  const std::tuple<TS_NODE_INT, TS_NODE_INT>& parents,
77  const TS_NODE_INT next_index)
78  {
79  if (breakpoints.empty())
80  {
81  this->push_back_edge(0., L, std::get<0>(parents),
82  next_index);
83  return;
84  }
85  auto itr = std::adjacent_find(std::begin(breakpoints),
86  std::end(breakpoints));
87  if (itr == std::end(breakpoints))
88  {
89  split_breakpoints_add_edges(breakpoints, parents,
90  next_index);
91  }
92  else
93  {
94  // Here, we need to reduce the input
95  // breakpoints to only those seen
96  // an odd number of times.
97  // Even numbers of the same breakpoint
98  // are "double x-overs" and thus
99  // cannot affect the genealogy.
100  std::vector<double> odd_breakpoints;
101  auto start = breakpoints.begin();
102  while (itr < breakpoints.end())
103  {
104  auto not_equal
105  = std::find_if(itr, breakpoints.end(),
106  [itr](const double d) {
107  return d != *itr;
108  });
109  int even = (std::distance(itr, not_equal) % 2
110  == 0.0);
111  odd_breakpoints.insert(odd_breakpoints.end(),
112  start, itr + 1 - even);
113  start = not_equal;
114  itr = std::adjacent_find(
115  start, std::end(breakpoints));
116  }
117  odd_breakpoints.insert(odd_breakpoints.end(), start,
118  breakpoints.end());
119  split_breakpoints_add_edges(odd_breakpoints, parents,
120  next_index);
121  }

186  void
187  sort_edges() noexcept
188  /// Sort the edge table. On PARENT birth times.
189  /// The sorting differs from msprime here. The difference
190  /// is that we assume that birth times are recorded forward in
191  /// time rather than backwards.
192  ///
193  /// In between simplifications, edges are appended to the table
194  /// starting at edge_offset. edges prior to that point are,
195  /// by definition, sorted b/c they are the output of simplification.
196  /// Thus, we only sort the new edges and then perform a left rotate
197  /// w.r.to edge_offset to put the nodes in the correct final order.
198  /// The rotation is exactly edge_table.size() swaps.
199  {
200  std::sort(edge_table.begin() + edge_offset, edge_table.end(),
201  [this](const edge& a, const edge& b) {
202  auto ga = this->node_table[a.parent].time;
203  auto gb = this->node_table[b.parent].time;
204  if (ga == gb)
205  {
206  if (a.parent == b.parent)
207  {
208  if (a.child == b.child)
209  {
210  return a.left < b.left;
211  }
212  return a.child < b.child;
213  }
214  return a.parent < b.parent;
215  }
216  return ga > gb;
217  });
218  if (edge_offset > 0)
219  {
220  std::rotate(edge_table.begin(),
221  edge_table.begin() + edge_offset,
222  edge_table.end());
223  }
224 #ifndef NDEBUG
225  // TODO: allow for exceptions
226  // rather than assertions.
227  if (edge_offset == 0)
228  {
229  assert(edges_are_sorted());
230  }
231  else if (preserved_nodes.empty())
232  {
233  assert(edges_are_sorted());
234  }
235  else
236  {
237  assert(edges_are_minimally_sorted());
238  }
239 #endif
240  }

419  void
420  build_indexes()
421  /// Generates the index vectors referred to
422  /// as I and O in Kelleher et al. (2016)
423  {
424  input_left.reserve(edge_table.size());
425  output_right.reserve(edge_table.size());
426  input_left.clear();
427  output_right.clear();
428  for (auto& e : edge_table)
429  {
430  assert(e.left < e.right);
431  input_left.emplace_back(e.left,
432  -node_table[e.parent].time,
433  e.parent, e.child);
434  output_right.emplace_back(e.right,
435  node_table[e.parent].time,
436  e.parent, e.child);
437  }
438  std::sort(input_left.begin(), input_left.end());
439  std::sort(output_right.begin(), output_right.end());
440  }
