112 /*
113  Note: this unit tests updates the same diploids at different steps.
114 
115  This is done so that the test is more complex than the intended use case,

123  pop.mutations.emplace_back(0.2,-0.1,0.0,0); //Selected, deleterious, recessive at position 0.2
124  pop.mutations.emplace_back(0.3,0.1,1.0,0); //Selected, beneficial, h=1 at position 0.3,
125  pop.mcounts.resize(3,0); //got make sure mcounts matches up in size. We init to zero b/c add_mutations will do the work for us.
126
