30  void
31  update_roots_outgoing(TS_NODE_INT p, TS_NODE_INT c,
32  marginal_tree& marginal)
33  // This is the algorithm used by tskit.
34  // The method is applied to nodes p and c
35  // AFTER the output index has been applied
36  // to a marginal tree.
37  //
38  // The criteria to mark something as a root
39  // are that its parent is NULL and it is
40  // above a sample. Here, c's parent is NULL
41  // because that is the result of processing
42  // "outgoing" nodes. Thus, c will be
43  // added to the list of nodes.
44  //
45  // The trick is to determine if the most
46  // ancient ancestor of p is NOT a root,
47  // which occurs if that node is not above any samples.
48  //
49  // While we are at it, we update above_sample.
50  // A node is above a sample if it is a sample
51  // or if any of its children are above a sample.
52  {
53  if (marginal.above_sample[c] == 1)
54  {
55  auto x = p;
56  auto root = x;
57  std::int8_t above_sample = 0;
58  while (x != TS_NULL_NODE && above_sample == 0)
59  {
60  // If this node is a sample, then
61  // it must descend from a root,
62  // which is why we OR this in the loop below.
63  above_sample = (marginal.sample_index_map[x]
64  != TS_NULL_NODE);
65  auto lc = marginal.left_child[x];
66  while (lc != TS_NULL_NODE && above_sample == 0)
67  {
68  above_sample
69  = above_sample
70  || marginal.above_sample[lc];
71  lc = marginal.right_sib[lc];
72  }
73  marginal.above_sample[x] = above_sample;
74  root = x;
75  x = marginal.parents[x];
76  }
77  // Now, root refers to the most ancient
78  // ancestor of p found in the above loop
79  if (above_sample == 0)
80  {
81  // Remove root from list of roots
82  auto lroot = marginal.left_sib[root];
83  auto rroot = marginal.right_sib[root];
84  marginal.left_root = TS_NULL_NODE;
85  if (lroot != TS_NULL_NODE)
86  {
87  marginal.right_sib[lroot] = rroot;
88  marginal.left_root = lroot;
89  }
90  if (rroot != TS_NULL_NODE)
91  {
92  marginal.left_sib[rroot] = lroot;
93  marginal.left_root = rroot;
94  }
95  marginal.left_sib[root] = TS_NULL_NODE;
96  marginal.right_sib[root] = TS_NULL_NODE;
97  }
98  if (marginal.left_root != TS_NULL_NODE)
99  {
100  //Put c into a pre-existing root list
101  auto lroot
102  = marginal.left_sib[marginal.left_root];
103  if (lroot != TS_NULL_NODE)
104  {
105  marginal.right_sib[lroot] = c;
106  }
107  marginal.left_sib[c] = lroot;
108  marginal.left_sib[marginal.left_root] = c;
109  }
110  marginal.right_sib[c] = marginal.left_root;
111  marginal.left_root = c;
112  }

115  void
116  update_roots_incoming(TS_NODE_INT p, TS_NODE_INT c,
117  TS_NODE_INT lsib, TS_NODE_INT rsib,
118  marginal_tree& marginal)
119  // This is the algorithm used by tskit. It is applied
120  // AFTER the incoming node list has updated marginal.
121  // Thus, p is parent to c.
122  //
123  // lsib and rsib are with respect to c BEFORE the incoming
124  // node list has updated marginal. This is a bit confusing:
125  // these values are used to remove c from the root list if
126  // necessary.
127  //
128  // NOTE: all values of c for which above_sample[c] == 1
129  // are in the root list.
130  {
131  if (marginal.above_sample[c])
132  {
133  auto x = p;
134  auto root = x;
135 
136  std::int8_t above_sample = 0;
137  while (x != TS_NULL_NODE && above_sample == 0)
138  {
139  above_sample = marginal.above_sample[x];
140  // c is above_sample and p is c's parent.
141  // Thus, all parents to p are above_sample, too.
142  marginal.above_sample[x] = 1;
143  root = x;
144  x = marginal.parents[x];
145  }
146  if (above_sample == 0)
147  // If we are here, then the above loop terminated
148  // by encountering a NULL node, because above_sample[x]
149  // must have been 0 for all x. However, because c is
150  // above sample, all nodes encountered have been update
151  // to be above_sample as well. Thus, the new value of root
152  // replaces c in the root list.
153  {
154  // Replace c with root in root list
155  if (lsib != TS_NULL_NODE)
156  {
157  marginal.right_sib[lsib] = root;
158  }
159  if (rsib != TS_NULL_NODE)
160  {
161  marginal.left_sib[rsib] = root;
162  }
163  marginal.left_sib[root] = lsib;
164  marginal.right_sib[root] = rsib;
165  marginal.left_root = root;
166  }
167  else
168  // If we are here, then we encountered a node
169  // ancestral to c where above_sample == 1.
170  // Thus, c can no longer be a root. If the current
171  // p is also a c in a later call to this function, then
172  // it may also be removed, etc..
173  {
174  // Remove c from root list
175  marginal.left_root = TS_NULL_NODE;
176  if (lsib != TS_NULL_NODE)
177  {
178  marginal.right_sib[lsib] = rsib;
179  marginal.left_root = lsib;
180  }
181  if (rsib != TS_NULL_NODE)
182  {
183  marginal.left_sib[rsib] = lsib;
184  marginal.left_root = rsib;
185  }
186  }
187  }
