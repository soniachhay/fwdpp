148  posmaker(const gsl_rng *r, const double &beg, const double &end,
149  lookup_table_t &lookup) const
150  {
151  double pos = gsl_ran_flat(r, beg, end);

160  /*!
161  Return a KTfwd::popgenmut
162 
163  \param r A gsl_rng
164  \param nmu Neutral mutation rate (per gamete, per generation)
165  \param smu Selected mutation rate (per gamete, per generation)
166  \param generation The current generation
167  \param recycling_bin A recycling bin for mutations
168  \param mutations A container of mutations
169  \param lookup Lookup table for mutations
170 
171  \precondition If nmu == 0, then nbeg/nend cannot be empty.
172  Similarly,
173  if smu == 0, then sbeg,end cannot be empty. These conditions are
174  checked in debug
175  mode via the assert macro. It is up to the calling environment
176  to prevent this situation
177  from arising. Also, nmu+smu must be > 0, and is also checked by
178  assert.
179  */
180  template <typename queue_t, typename lookup_table_t,
181  typename mcont_t>
182  inline result_type
183  make_mut(queue_t &recycling_bin, mcont_t &mutations,
184  const gsl_rng *r, const double &nmu, const double &smu,
185  unsigned generation, lookup_table_t &lookup) const
186  {
187  assert(nmu + smu > 0.);
188  bool is_neutral
189  = (gsl_rng_uniform(r) < nmu / (nmu + smu)) ? true : false;
190  if (is_neutral)
191  {
192  assert(!nbeg.empty());
193  assert(!nend.empty());
194  size_t region = gsl_ran_discrete(r, nlookup.get());
195  double pos
196  = posmaker(r, nbeg[region], nend[region], lookup);
197  return fwdpp_internal::recycle_mutation_helper(
198  recycling_bin, mutations, pos, 0., 0., generation,
199  nlabels[region]);
200  }
201  assert(!sbeg.empty());
202  assert(!send.empty());
203  size_t region = gsl_ran_discrete(r, slookup.get());
204  double pos = posmaker(r, sbeg[region], send[region], lookup);
205  return fwdpp_internal::recycle_mutation_helper(

266  /*!
267  Returns a position from a region that is chosen based on region
268  weights.
269 
270  \precondition If recrate == 0, then beg and end cannot be empty.
271  It is up to the calling environment to make sure
272  that this cannot happen. This is checked in debug mode via the
273  assert macro.
274  */
275  template <typename gamete_t, typename mcont_t>
276  inline result_type
277  operator()(const gsl_rng *r, const double recrate,
278  const gamete_t &, const gamete_t &,
279  const mcont_t &) const
280  {
281  assert(!(recrate == 0. && (beg.empty() || end.empty())));
282  auto nbreaks = gsl_ran_poisson(r, recrate);
283  if (!nbreaks)
284  return {};
285 
286  result_type rv;
287  for (unsigned i = 0; i < nbreaks; ++i)
288  {
289  size_t region = gsl_ran_discrete(r, lookup.get());
290  rv.push_back(
291  gsl_ran_flat(r, beg[region], end[region]));
292  }
293  std::sort(rv.begin(), rv.end());
294  rv.push_back(std::numeric_limits<double>::max());
295  return rv;
296  }
297  };
