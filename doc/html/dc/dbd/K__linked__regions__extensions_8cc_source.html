24  template <typename diploid_t, typename gcont_t, typename mcont_t>
25  inline double
26  operator()(const diploid_t &dip, const gcont_t &haploid_genomes,
27  const mcont_t &mutations, const unsigned K) const noexcept
28  /*
29  The fitness model is additive over loci, multiplicative within loci
30  */
31  {
32  double rv = 0.0;
33  auto &g1 = haploid_genomes[dip.first];
34  auto &g2 = haploid_genomes[dip.second];
35  for (unsigned i = 0; i < K; ++i)
36  {
37  /*
38  Find the iterators corrsponding to locus 1 in g1 and g2.
39  Because haploid_genome store mutation keys sorted by mutation
40  position,
41  we can use efficient binary searches.
42 
43  We want the positions of all mutations such that i <= p <
44  i+1,
45  which correspond to lower_bound and upper_bound,
46  respectively.
47  */
48  auto stop1 = std::upper_bound(
49  g1.smutations.begin(), g1.smutations.end(), double(i + 1),
50  [&mutations](const double val, const std::size_t i) {
51  return val < mutations[i].pos;
52  });
53  auto start1 = std::lower_bound(
54  g1.smutations.begin(), stop1, double(i),
55  [&mutations](const std::size_t i, const double val) {
56  return mutations[i].pos < val;
57  });
58  auto stop2 = std::upper_bound(
59  g2.smutations.begin(), g2.smutations.end(), double(i + 1),
60  [&mutations](const double val, const std::size_t i) {
61  return val < mutations[i].pos;
62  });
63  auto start2 = std::lower_bound(
64  g2.smutations.begin(), stop2, double(i),
65  [&mutations](const std::size_t i, const double val) {
66  return mutations[i].pos < val;
67  });
68 
69  rv += fwdpp::multiplicative_diploid(fwdpp::fitness(2.0))(
70  start1, stop1, start2, stop2, mutations)
71  - 1.0;
72  }
73  return std::max(1.0 + rv, 0.0);
74  }
75 };
