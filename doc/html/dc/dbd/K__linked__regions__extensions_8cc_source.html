23  template <typename diploid_t, typename gcont_t, typename mcont_t>
24  inline double
25  operator()(const diploid_t &dip, const gcont_t &gametes,
26  const mcont_t &mutations, const unsigned K) const noexcept
27  /*
28  The fitness model is additive over loci, multiplicative within loci
29  */
30  {
31  double rv = 0.0;
32  auto &g1 = gametes[dip.first];
33  auto &g2 = gametes[dip.second];
34  for (unsigned i = 0; i < K; ++i)
35  {
36  /*
37  Find the iterators corrsponding to locus 1 in g1 and g2.
38  Because gamete store mutation keys sorted by mutation
39  position,
40  we can use efficient binary searches.
41 
42  We want the positions of all mutations such that i <= p <
43  i+1,
44  which correspond to lower_bound and upper_bound,
45  respectively.
46  */
47  auto stop1 = std::upper_bound(
48  g1.smutations.begin(), g1.smutations.end(), double(i + 1),
49  [&mutations](const double val, const std::size_t i) {
50  return val < mutations[i].pos;
51  });
52  auto start1 = std::lower_bound(
53  g1.smutations.begin(), stop1, double(i),
54  [&mutations](const std::size_t i, const double val) {
55  return mutations[i].pos < val;
56  });
57  auto stop2 = std::upper_bound(
58  g2.smutations.begin(), g2.smutations.end(), double(i + 1),
59  [&mutations](const double val, const std::size_t i) {
60  return val < mutations[i].pos;
61  });
62  auto start2 = std::lower_bound(
63  g2.smutations.begin(), stop2, double(i),
64  [&mutations](const std::size_t i, const double val) {
65  return mutations[i].pos < val;
66  });
67 
68  rv += fwdpp::multiplicative_diploid(2.0)(start1, stop1, start2,
69  stop2, mutations)
70  - 1.0;
71  }
72  return std::max(1.0 + rv, 0.0);
73  }
74 };
