100  template <typename mutation_model, typename diploid_t, typename gcont_t,
101  typename mcont_t>
102  std::vector<uint_t>
103  generate_new_mutations(flagged_mutation_queue &recycling_bin,
104  const gsl_rng *r, const double &mu,
105  const diploid_t &dip, gcont_t &haploid_genomes,
106  mcont_t &mutations, const std::size_t g,
107  const mutation_model &mmodel)
108  ///
109  /// Return a vector of keys to new mutations. The keys
110  /// will be sorted according to mutation postition.
111  ///
112  /// \param recycling_bin The queue for recycling mutations
113  /// \param r A random number generator
114  /// \param mu The total mutation rate
115  /// \param dip A single-locus diploid
116  /// \param haploid_genomes Vector of haploid_genomes
117  /// \param mutations Vector of mutations
118  /// \param g index of haploid_genome to mutate
119  /// \param mmodel The mutation policy
120  ///
121  /// \return Vector of mutation keys, sorted according to position
122  ///
123  {
124  unsigned nm = gsl_ran_poisson(r, mu);
125  std::vector<uint_t> rv;
126  rv.reserve(nm);
127  for (unsigned i = 0; i < nm; ++i)
128  {
129  rv.emplace_back(fwdpp_internal::mmodel_dispatcher(
130  mmodel, dip, haploid_genomes[g], mutations,
131  recycling_bin));
132  }
133  std::sort(rv.begin(), rv.end(),
134  [&mutations](const uint_t a, const uint_t b) {
135  return mutations[a].pos < mutations[b].pos;
136  });
137  return rv;
138  }

176  template <typename gcont_t, typename mcont_t, typename queue_type,
177  typename new_mutations_type, typename breakpoints_type>
178  uint_t
179  mutate_recombine(
180  const new_mutations_type &new_mutations,
181  const breakpoints_type &breakpoints, const std::size_t g1,
182  const std::size_t g2, gcont_t &haploid_genomes, mcont_t &mutations,
183  queue_type &haploid_genome_recycling_bin,
184  typename gcont_t::value_type::mutation_container &neutral,
185  typename gcont_t::value_type::mutation_container &selected)
186  ///
187  /// Update apply new mutations and recombination events to
188  /// an offspring's haploid_genome.
189  ///
190  /// \param new_mutations A range referring to new mutation keys. Must be traversable via non-member begin/end
191  /// functions using argument-dependent lookup (ADL).
192  /// \param breakpoints A range of recombination breakpoints traversable via non-member begin/end functions.
193  /// \param g1 Parental haploid_genome 1
194  /// \param g2 Parental haploid_genome 2
195  /// \param haploid_genomes The vector of haploid_genomes in the population
196  /// \param mutations The vector of mutations in the population
197  /// \param haploid_genome_recycling_bin FIFO queue for haploid_genome recycling
198  /// \param neutral Temporary container for updating neutral mutations
199  /// \param selected Temporary container for updatng selected positions
200  ///
201  /// \return The index of the new offspring haploid_genome in \a haploid_genomes.
202  ///
203  /// TODO: unit test this note. I simply cannot believe it! :)
204  /// \note For efficiency, it is helpful if \a new_mutations is sorted
205  /// by mutation position. fwdpp::generate_new_mutations exists to help in
206  /// that
207  /// regard. Many of the evolve functions used in this library and other
208  /// packages by the author will use fwdpp::generate_breakpoints to
209  /// generate \a breakpoints. That is not, however, required.
210  ///
211  /// \todo Need a unit test on what happens if \a mutation_keys is not a sorted range
212  ///
213  /// \version
214  /// This function was added in fwdpp 0.5.7.
215  {
216  if (begin(new_mutations) == end(new_mutations)
217  && begin(breakpoints) == end(breakpoints))
218  {
219  return g1;
220  }
221  else if (begin(breakpoints)
222  == end(breakpoints)) // only mutations to deal with
223  {
224  fwdpp_internal::prep_temporary_containers(
225  g1, g2, haploid_genomes, neutral, selected);
226  auto nb = haploid_genomes[g1].mutations.begin(),
227  sb = haploid_genomes[g1].smutations.begin();
228  const auto ne = haploid_genomes[g1].mutations.end(),
229  se = haploid_genomes[g1].smutations.end();
230  for (auto m = begin(new_mutations); m < end(new_mutations);
231  ++m)
232  {
233  if (mutations[*m].neutral)
234  {
235  nb = fwdpp_internal::insert_new_mutation(
236  nb, ne, *m, mutations, neutral);
237  }
238  else
239  {
240  sb = fwdpp_internal::insert_new_mutation(
241  sb, se, *m, mutations, selected);
242  }
243  }
244  neutral.insert(neutral.end(), nb, ne);
245  selected.insert(selected.end(), sb, se);
246 
247 #ifndef NDEBUG
248  std::size_t new_neutral = 0, new_selected = 0;
249  for (auto m : new_mutations)
250  {
251  if (mutations[m].neutral)
252  {
253  ++new_neutral;
254  }
255  if (!mutations[m].neutral)
256  {
257  ++new_selected;
258  }
259  }
260  if (neutral.size()
261  != haploid_genomes[g1].mutations.size() + new_neutral)
262  {
263  throw std::runtime_error("FWDPP DEBUG: failure to add "
264  "all new neutral mutations");
265  }
266  if (selected.size()
267  != haploid_genomes[g1].smutations.size() + new_selected)
268  {
269  throw std::runtime_error("FWDPP DEBUG: failure to add "
270  "all new selected mutations");
271  }
272 #endif
273  return recycle_haploid_genome(haploid_genomes,
274  haploid_genome_recycling_bin,
275  neutral, selected);
276  }
277  if (breakpoints.size() == 1)
278  {
279  throw std::runtime_error(
280  "invalid number of breakpoints. likely sentinel error");
281  }
282  // If we get here, there are mutations and
283  // recombinations to handle
284  fwdpp_internal::prep_temporary_containers(g1, g2, haploid_genomes,
285  neutral, selected);
286 
287  auto itr = haploid_genomes[g1].mutations.cbegin();
288  auto jtr = haploid_genomes[g2].mutations.cbegin();
289  auto itr_s = haploid_genomes[g1].smutations.cbegin();
290  auto jtr_s = haploid_genomes[g2].smutations.cbegin();
291  auto itr_e = haploid_genomes[g1].mutations.cend();
292  auto itr_s_e = haploid_genomes[g1].smutations.cend();
293  auto jtr_e = haploid_genomes[g2].mutations.cend();
294  auto jtr_s_e = haploid_genomes[g2].smutations.cend();
295 
296  auto next_mutation = begin(new_mutations);
297  for (auto i = begin(breakpoints); i != end(breakpoints);)
298  {
299  if (next_mutation != end(new_mutations)
300  && mutations[*next_mutation].pos < *i)
301  {
302  const auto mut = &mutations[*next_mutation];
303  itr = fwdpp_internal::rec_gam_updater(
304  itr, itr_e, mutations, neutral, mut->pos);
305  itr_s = fwdpp_internal::rec_gam_updater(
306  itr_s, itr_s_e, mutations, selected, mut->pos);
307  jtr = fwdpp_internal::rec_update_itr(
308  jtr, jtr_e, mutations, mut->pos);
309  jtr_s = fwdpp_internal::rec_update_itr(
310  jtr_s, jtr_s_e, mutations, mut->pos);
311  if (mut->neutral)
312  {
313  neutral.push_back(*next_mutation);
314  }
315  else
316  {
317  selected.push_back(*next_mutation);
318  }
319  ++next_mutation;
320  }
321  else
322  {
323  itr = fwdpp_internal::rec_gam_updater(
324  itr, itr_e, mutations, neutral, *i);
325  itr_s = fwdpp_internal::rec_gam_updater(
326  itr_s, itr_s_e, mutations, selected, *i);
327  jtr = fwdpp_internal::rec_update_itr(jtr, jtr_e,
328  mutations, *i);
329  jtr_s = fwdpp_internal::rec_update_itr(jtr_s, jtr_s_e,
330  mutations, *i);
331  std::swap(itr, jtr);
332  std::swap(itr_s, jtr_s);
333  std::swap(itr_e, jtr_e);
334  std::swap(itr_s_e, jtr_s_e);
335  ++i;
336  }
337  }
338 #ifndef NDEBUG
339  if (next_mutation != end(new_mutations))
340  {
341  throw std::runtime_error("FWDPP DEBUG: fatal error during "
342  "mutation/recombination");
343  }
344 #endif
345  return recycle_haploid_genome(
346  haploid_genomes, haploid_genome_recycling_bin, neutral, selected);
347  }

353  const gsl_rng *r, gcont_t &haploid_genomes, mcont_t &mutations,
354  std::tuple<std::size_t, std::size_t, std::size_t, std::size_t>
355  parental_haploid_genomes,
356  const recmodel &rec_pol, const mutmodel &mmodel, const double mu,
357  flagged_haploid_genome_queue &haploid_genome_recycling_bin,
358  flagged_mutation_queue &mutation_recycling_bin, diploid_t &dip,
359  typename gcont_t::value_type::mutation_container &neutral,
360  typename gcont_t::value_type::mutation_container &selected)
361  ///
362  /// "Convenience" function for generating offspring haploid_genomes.
363  ///
364  /// This function calls fwdpp::generate_breakpoints,
365  /// fwdpp::generate_new_mutations, and fwdpp::mutate_recombine,
366  /// resulting in two offspring haploid_genomes.
367  ///
368  /// \param r A gsl_rng *
369  /// \param haploid_genomes Vector of haploid_genomes in population
370  /// \param mutations Vector of mutations in population
371  /// \param parental_haploid_genomes Tuple of haploid_genome keys for each parent
372  /// \param rec_pol Policy to generate recombination breakpoints
373  /// \param mmodel Policy to generate new mutations
374  /// \param mu Total mutation rate (per haploid_genome).
375  /// \param haploid_genome_recycling_bin FIFO queue for haploid_genome recycling
376  /// \param mutation_recycling_bin FIFO queue for mutation recycling
377  /// \param dip The offspring
378  /// \param neutral Temporary container for updating neutral mutations
379  /// \param selected Temporary container for updating selected mutations
380  ///
381  /// \return Number of recombination breakpoints and mutations for each
382  /// haploid_genome.
383  ///
384  /// \note
385  /// \a parental_haploid_genomes should contain parent one/haploid_genome one,
386  /// parent one/haploid_genome two, parent two/haploid_genome one,
387  /// and parent two/haploid_genome two, in that order.
388  ///
389  /// \version
390  /// Added in fwdpp 0.5.7.
391  {
392  auto p1g1 = std::get<0>(parental_haploid_genomes);
393  auto p1g2 = std::get<1>(parental_haploid_genomes);
394  auto p2g1 = std::get<2>(parental_haploid_genomes);
395  auto p2g2 = std::get<3>(parental_haploid_genomes);
396  // Now, we generate the crossover breakpoints for
397  // both parents,as well as the new mutations that we'll place
398  // onto each haploid_genome. The specific order of operations below
399  // is done to ensure the exact same output as fwdpp 0.5.6 and
400  // earlier.
401  // The breakpoints are of type std::vector<double>, and
402  // the new_mutations are std::vector<fwdpp::uint_t>, with
403  // the integers representing the locations of the new mutations
404  // in "mutations".
405 
406  auto breakpoints = generate_breakpoints(
407  dip, p1g1, p1g2, haploid_genomes, mutations, rec_pol);
408  auto breakpoints2 = generate_breakpoints(
409  dip, p2g1, p2g2, haploid_genomes, mutations, rec_pol);
410  auto new_mutations
411  = generate_new_mutations(mutation_recycling_bin, r, mu, dip,
412  haploid_genomes, mutations, p1g1, mmodel);
413  auto new_mutations2
414  = generate_new_mutations(mutation_recycling_bin, r, mu, dip,
415  haploid_genomes, mutations, p2g1, mmodel);
416 
417  // Pass the breakpoints and new mutation keys on to
418  // fwdpp::mutate_recombine (defined in
419  // fwdpp/mutate_recombine.hpp),
420  // which splices together the offspring haploid_genome and returns its
421  // location in haploid_genomes. The location of the offspring haploid_genome
422  // is
423  // either reycled from an extinct haploid_genome or it is the location
424  // of a
425  // new haploid_genome emplace_back'd onto the end.
426  dip.first = mutate_recombine(
427  new_mutations, breakpoints, p1g1, p1g2, haploid_genomes, mutations,
428  haploid_genome_recycling_bin, neutral, selected);
429  debug::haploid_genome_is_sorted(haploid_genomes[dip.first], mutations);
430  dip.second = mutate_recombine(
431  new_mutations2, breakpoints2, p2g1, p2g2, haploid_genomes,
432  mutations, haploid_genome_recycling_bin, neutral, selected);
433  debug::haploid_genome_is_sorted(haploid_genomes[dip.second],
434  mutations);
435  haploid_genomes[dip.first].n++;
436  haploid_genomes[dip.second].n++;
437 
438  debug::haploid_genome_is_extant(haploid_genomes[dip.first]);
439  debug::haploid_genome_is_extant(haploid_genomes[dip.second]);
440 
441  auto nrec = (!breakpoints.empty()) ? breakpoints.size() - 1 : 0;
442  auto nrec2 = (!breakpoints2.empty()) ? breakpoints2.size() - 1 : 0;
443  return std::make_tuple(nrec, nrec2, new_mutations.size(),
444  new_mutations2.size());
445  }
