20  struct state_matrix
21  /*! \brief Simplistic matrix representation of mutations
22  * This type uses std::vector<std::int8_t> to hold a matrix
23  * representing the genotypes for a set of diploids.
24  *
25  * For a haplotype matrix of n individuals, the data represent
26  * 2n columns with a 0/1 encoding representing ancestral/derived.
27  *
28  * For a genotype matrix of n individuals, the data represent
29  * n columns with a 0/1/2 encoding for the number of copies of the
30  * derived mutation.
31  *
32  * The data layout is row-major (aka "C-style") ordering,
33  * facilitating compatibility with GSL matrix types,
34  * NumPy 2D arrays, etc. Note that GSL matrices may be
35  * constructed using gsl_matrix_view_array or
36  * gsl_matrix_const_view_array for cases where a matrix of
37  * only neutral or only selected mutations is needed.
38  *
39  * We use the 8-bit integer type to save space. In practice,
40  * one may convert (via copy) to other types for operations like
41  * regression.
42  *
43  * \note This type is not constructed directly, but rather returned
44  * by other functions.
45  *
46  * \ingroup samplingPops
47  */
48  {
49  //! The state data
50  std::vector<std::int8_t> data;
51  //! Positions of variable sites
52  std::vector<double> positions;
53  //! Construct an empty object
54  state_matrix() : data(), positions() {}
55  //! Perfect-forwarding constructor
56  template <typename T, typename P>
57  state_matrix(T &&t, P &&p)
58  : data(std::forward<T>(t)), positions(std::forward<P>(p))
59  {
60  }
61  };

114  const bool include_neutral, const bool include_selected)
115  /*!
116  * For a sample defined by a set of diploids, obtain the keys corresponding
117  * to all mutations in that sample.
118  *
119  * \param pop The population object
120  * \param individuals The indexes of individuals in the sample
121  * \param include_neutral If true, obtain keys for neutral sites
122  * \param include_selected If true, obtain keys for selected sites
123  *
124  * \return A pair of vectors representing neutral and selected keys,
125  * respectively. The value
126  * type of each vector is std::pair<std::size_t,fwdpp::uint_t>. The first
127  * value is the mutation
128  * key, and the second is its frequency in the sample.
129  *
130  * Several comments are required:
131  *
132  * 1. Return values are unsorted with respect to anything meaningful. If
133  * you wish to sort on position,
134  * etc., do so yourself.
135  * 2. In general, any manipulation of the keys is possible. For example,
136  * removing mutations that are fixed
137  * in the sample, or whose frequency is or is not within some desired
138  * range, is easily doable via the
139  * "erase/remove" idiom.
140  * 3. Keys from multiple samples can be merged to form new vectors where
141  * the key elements are unique and
142  * the frequencies are summed, all using standard C++.
143  *
144  * \ingroup samplingPops
145  */
146  {
147  return data_matrix_details::mutation_keys(
148  pop.diploids, individuals, pop.haploid_genomes, pop.mcounts,
149  include_neutral, include_selected, typename poptype::popmodel_t());
150  }

153  inline void
154  sort_keys(const mcont_t &mutations,
155  std::vector<std::pair<std::size_t, uint_t>> &keys)
156  /*! \brief Sort keys by position
157  *
158  * Takes the data returned from mutation_keys and sorts
159  * it by increasing mutation positions.
160  *
161  * \param mutations A mutation container
162  * \param keys Returned from mutation_keys
163  *
164  * \ingroup samplingPops
165  */
166  {
167  const auto comp
168  = [&mutations](const std::pair<std::size_t, uint_t> &a,
169  const std::pair<std::size_t, uint_t> &b) {
170  return mutations[a.first].pos < mutations[b.first].pos;
171  };
172  std::sort(keys.begin(), keys.end(), comp);
173  }
