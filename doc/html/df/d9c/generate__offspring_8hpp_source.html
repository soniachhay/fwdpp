162  mutation_policy, genetics.mutation_recycling_bin,
163  genetics.haploid_genome_recycling_bin, genetics.neutral,
164  genetics.selected, pop);

182  template <typename genetic_param_holder,
183  typename mutation_handling_policy, typename poptype>
184  std::pair<mut_rec_intermediates, mut_rec_intermediates>
185  generate_offspring(const gsl_rng* r,
186  const std::pair<std::size_t, std::size_t> parents,
187  const mutation_handling_policy& mutation_policy,
188  poptype& pop, genetic_param_holder& genetics,
189  typename poptype::diploid_t& offspring)
190  /// \brief Generate offspring haploid_genomes and return breakpoints plus mutation keys
191  ///
192  /// \param r Random number generator
193  /// \param parents Indexes of the offspring parents in \a pop
194  /// \param mutation_policy Either all_mutations or selected_variants_only. See below.
195  /// \param pop fwdpp::poptypes::diploid_population
196  /// \param genetics A duck type of fwdpp::genetic_parameters.
197  /// \param offspring The offspring for which we will generate haploid_genomes.
198  ///
199  /// The final three parameters will be modified.
200  ///
201  /// \return A pair of mut_rec_intermediates, corresponding to what is passed on from
202  /// each parent.
203  ///
204  /// The operations are dispatched out to functions based on the population type. These
205  /// functions make calls to fwdpp::mutate_recombine.
206  ///
207  /// The parameter \a mutation_policy governs what types of mutations are entered
208  /// into the haploid_genomes of \a offspring. If the policy is fwdpp::ts::all_mutations, then
209  /// neutral and selected variants are placed in an offspring's haploid_genomes. If the policy is
210  /// fwdpp::ts::selected_variants_only, then only selected mutations are placed into the haploid_genomes.
211  /// Regardless of the policy, ALL mutations are contained in the return value, with the idea
212  /// that the caller will record them into a fwdpp::ts::table_collection.
213  ///
214  /// \note The function type genetic_param_holder::generate_mutations must return
215  /// std::vector<fwdpp::uint_t>, with the values reflecting the locations of new
216  /// mutations on \a pop.mutations. Further, this function must already bind
217  /// any relevant mutation rates (as this generate_offspring does not accept them
218  /// as arguments).
219  ///
220  /// \version 0.7.4 Added to fwdpp::ts.
221  ///
222  {
223  return detail::generate_offspring_details(
224  typename poptype::popmodel_t(), r, parents, mutation_policy,
225  pop, genetics, offspring);
226  }
