217 SimData merge( const std::vector<std::pair<double,std::string> > & sample1,
218  const std::vector<std::pair<double,std::string> > & sample2 ,
219  const unsigned & nsam)
220 {
221  std::map<double, std::string> temp;
222 
223  for( unsigned i=0;i<sample1.size();++i)
224  {
225  temp[sample1[i].first] = std::string(sample1[i].second + std::string(nsam,'0'));
226  }
227 
228  for( unsigned i=0;i<sample2.size();++i)
229  {
230  std::map<double,std::string>::iterator itr = temp.find(sample2[i].first);
231  if( itr == temp.end() )
232  {
233  temp[sample2[i].first] = std::string( std::string(nsam,'0') + sample2[i].second );
234  }
235  else
236  {
237  std::copy( sample2[i].second.begin(),sample2[i].second.end(),itr->second.begin()+nsam );
238  }
239  }
240  std::vector<std::pair<double,std::string> > rv( temp.begin(),temp.end() );
241  std::sort(rv.begin(),rv.end(),
242  [](std::pair<double,std::string> lhs,

244  return SimData(rv.begin(),rv.end());
