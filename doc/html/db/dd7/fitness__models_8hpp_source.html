42 /*!
43  \defgroup fitness Policies for calculating fitnesses and trait values.
44  This group contains the following data structures to help you implement
45  custom fitness policies:
46  1. KTfwd::site_dependent_genetic_value for implementing typical
47  population-genetic-like models where trait values are a function of the
48  properties of individual mutations
49  2. KTfwd::haplotype_dependent_fitness for implementing models where trait
50  values are the functions properties of haplotypes.
51 
52  Truthfully, latter is so trivial that a library user may never see any need
53  for it.
54 
55  The library also defines two site-dependent fitness models, which may also
56  be used for trait value calculations:
57  1. KTfwd::multiplicative_diploid
58  2. KTfwd::additive_diploid
59 
60  These are arguably the "standard" models of the field as far as selection is
61  concerned.
62 
63  Finally, KTfwd::no_selection is provided to force all diploid fitnesses to be
64  equal to 1.

130  /*!
131  Range-based call operator. Calculates genetic values over ranges of
132  mutation keys first1/last1
133  and first2/last2, which are iterators derived from the 'smutations'
134  of two gametes in a diploid.
135 
136  \param first1 Iterator to first mutation derived from gamete 1
137  \param last1 Iterator to one past the last mutation derived from
138  gamete 1
139  \param first2 Iterator to first mutation derived from gamete 2
140  \param last2 Iterator to one past the last mutation derived from
141  gamete 2
142  \param mutations The container of mutations for the simulation
143  \param fpol_hom Policy that updates genetic value for a homozygous
144  mutation.
145  \param fpol_het Policy that updates genetic value for a heterozygous
146  mutation.
147  \param starting_value The initial genetic value.
148 
149  \returns Fitness (double)
150  */
151  {
152  static_assert(
153  traits::is_mutation<typename mcont_t::value_type>::value,
154  "mcont_t::value_type must be a mutation type");
155  static_assert(
156  std::is_convertible<updating_policy_hom,
157  std::function<void(
158  double &, const typename mcont_t::
159  value_type &)>>::value,
160  "decltype(fpol_hom) must be convertible to "
161  "std::function<void(double &,const typename "
162  "mcont_t::value_type");
163  static_assert(
164  std::is_convertible<updating_policy_het,
165  std::function<void(
166  double &, const typename mcont_t::
167  value_type &)>>::value,
168  "decltype(fpol_het) must be convertible to "
169  "std::function<void(double &,const typename "
170  "mcont_t::value_type");
171  result_type w = starting_value;
172  if (first1 == last1 && first2 == last2)
173  return w;
174  else if (first1 == last1)
175  {
176  for (; first2 != last2; ++first2)
177  fpol_het(w, mutations[*first2]);
178  return w;
179  }
180  else if (first2 == last2)
181  {
182  for (; first1 != last1; ++first1)
183  fpol_het(w, mutations[*first1]);
184  return w;
185  }
186  for (; first1 != last1; ++first1)
187  {
188  for (;
189  first2 != last2 && *first1 != *first2
190  && !(mutations[*first2].pos > mutations[*first1].pos);
191  ++first2)
192  // All mutations in this range are Aa
193  {
194  assert(mutations[*first2].pos
195  < mutations[*first1].pos);
196  fpol_het(w, mutations[*first2]);
197  }
198  if (first2 < last2 && *first1 == *first2) // mutation with
199  // index first1
200  // is homozygous
201  {
202  assert(mutations[*first2].pos
203  == mutations[*first1].pos);
204  fpol_hom(w, mutations[*first1]);
205  ++first2; // increment so that we don't re-process
206  // this site as a het next time 'round
207  }
208  else // mutation first1 is heterozygous
209  {
210  fpol_het(w, mutations[*first1]);
211  }
212  }
213  for (; first2 != last2; ++first2)
214  {
215  fpol_het(w, mutations[*first2]);
216  }
217  return w;
218  }
