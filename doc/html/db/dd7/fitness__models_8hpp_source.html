44 /*!
45  \defgroup fitness Policies for calculating fitnesses and trait values.
46  This group contains the following data structures to help you implement
47  custom fitness policies:
48  1. fwdpp::site_dependent_genetic_value for implementing typical
49  population-genetic-like models where trait values are a function of the
50  properties of individual mutations
51  2. fwdpp::haplotype_dependent_fitness for implementing models where trait
52  values are the functions properties of haplotypes.
53 
54  Truthfully, latter is so trivial that a library user may never see any need
55  for it.
56 
57  The library also defines two site-dependent fitness models, which may also
58  be used for trait value calculations:
59  1. fwdpp::multiplicative_diploid
60  2. fwdpp::additive_diploid
61 
62  These are arguably the "standard" models of the field as far as selection is
63  concerned.
64 
65  Finally, fwdpp::no_selection is provided to force all diploid fitnesses to be
66  equal to 1.
67 */

128  inline result_type
129  operator()(iterator_t first1, iterator_t last1, iterator_t first2,
130  iterator_t last2, const mcont_t &mutations,
131  const updating_policy_hom &fpol_hom,
132  const updating_policy_het &fpol_het,

134  /*!
135  Range-based call operator. Calculates genetic values over ranges of
136  mutation keys first1/last1
137  and first2/last2, which are iterators derived from the 'smutations'
138  of two haploid_genomes in a diploid.
139 
140  \param first1 Iterator to first mutation derived from haploid_genome 1
141  \param last1 Iterator to one past the last mutation derived from
142  haploid_genome 1
143  \param first2 Iterator to first mutation derived from haploid_genome 2
144  \param last2 Iterator to one past the last mutation derived from
145  haploid_genome 2
146  \param mutations The container of mutations for the simulation
147  \param fpol_hom Policy that updates genetic value for a homozygous
148  mutation.
149  \param fpol_het Policy that updates genetic value for a heterozygous
150  mutation.
151  \param starting_value The initial genetic value.
152 
153  \returns Fitness (double)
154  */
155  {
156  static_assert(
157  traits::is_mutation<typename mcont_t::value_type>::value,
158  "mcont_t::value_type must be a mutation type");
159  static_assert(
160  std::is_convertible<
161  updating_policy_hom,
162  std::function<void(double &, const typename mcont_t::
163  value_type &)>>::value,
164  "decltype(fpol_hom) must be convertible to "
165  "std::function<void(double &,const typename "
166  "mcont_t::value_type");
167  static_assert(
168  std::is_convertible<
169  updating_policy_het,
170  std::function<void(double &, const typename mcont_t::
171  value_type &)>>::value,
172  "decltype(fpol_het) must be convertible to "
173  "std::function<void(double &,const typename "
174  "mcont_t::value_type");
175  result_type w = starting_value;
176  if (first1 == last1 && first2 == last2)
177  return w;
178  else if (first1 == last1)
179  {
180  for (; first2 != last2; ++first2)
181  fpol_het(w, mutations[*first2]);
182  return w;
183  }
184  else if (first2 == last2)
185  {
186  for (; first1 != last1; ++first1)
187  fpol_het(w, mutations[*first1]);
188  return w;
189  }
190  for (; first1 != last1; ++first1)
191  {
192  for (; first2 != last2 && *first1 != *first2
193  && mutations[*first2].pos < mutations[*first1].pos;
194  ++first2)
195  // All mutations in this range are Aa
196  {
197  fpol_het(w, mutations[*first2]);
198  }
199  if (first2 < last2
200  && (*first1 == *first2
201  || mutations[*first1].pos
202  == mutations[*first2].pos))
203  // mutation with index first1 is homozygous
204  {
205  fpol_hom(w, mutations[*first1]);
206  ++first2; // increment so that we don't re-process
207  // this site as a het next time 'round
208  }
209  else // mutation first1 is heterozygous
210  {
211  fpol_het(w, mutations[*first1]);
212  }
213  }
214  for (; first2 != last2; ++first2)
215  {
216  fpol_het(w, mutations[*first2]);
217  }
218  return w;
219  }
