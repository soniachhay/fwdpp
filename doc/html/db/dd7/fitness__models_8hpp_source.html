43 /*!
44  \defgroup fitness Policies for calculating fitnesses and trait values.
45  This group contains the following data structures to help you implement
46  custom fitness policies:
47  1. fwdpp::site_dependent_genetic_value for implementing typical
48  population-genetic-like models where trait values are a function of the
49  properties of individual mutations
50  2. fwdpp::haplotype_dependent_fitness for implementing models where trait
51  values are the functions properties of haplotypes.
52 
53  Truthfully, latter is so trivial that a library user may never see any need
54  for it.
55 
56  The library also defines two site-dependent fitness models, which may also
57  be used for trait value calculations:
58  1. fwdpp::multiplicative_diploid
59  2. fwdpp::additive_diploid
60 
61  These are arguably the "standard" models of the field as far as selection is
62  concerned.
63 
64  Finally, fwdpp::no_selection is provided to force all diploid fitnesses to be
65  equal to 1.
66 */

131  /*!
132  Range-based call operator. Calculates genetic values over ranges of
133  mutation keys first1/last1
134  and first2/last2, which are iterators derived from the 'smutations'
135  of two gametes in a diploid.
136 
137  \param first1 Iterator to first mutation derived from gamete 1
138  \param last1 Iterator to one past the last mutation derived from
139  gamete 1
140  \param first2 Iterator to first mutation derived from gamete 2
141  \param last2 Iterator to one past the last mutation derived from
142  gamete 2
143  \param mutations The container of mutations for the simulation
144  \param fpol_hom Policy that updates genetic value for a homozygous
145  mutation.
146  \param fpol_het Policy that updates genetic value for a heterozygous
147  mutation.
148  \param starting_value The initial genetic value.
149 
150  \returns Fitness (double)
151  */
152  {
153  static_assert(
154  traits::is_mutation<typename mcont_t::value_type>::value,
155  "mcont_t::value_type must be a mutation type");
156  static_assert(
157  std::is_convertible<
158  updating_policy_hom,
159  std::function<void(double &, const typename mcont_t::
160  value_type &)>>::value,
161  "decltype(fpol_hom) must be convertible to "
162  "std::function<void(double &,const typename "
163  "mcont_t::value_type");
164  static_assert(
165  std::is_convertible<
166  updating_policy_het,
167  std::function<void(double &, const typename mcont_t::
168  value_type &)>>::value,
169  "decltype(fpol_het) must be convertible to "
170  "std::function<void(double &,const typename "
171  "mcont_t::value_type");
172  result_type w = starting_value;
173  if (first1 == last1 && first2 == last2)
174  return w;
175  else if (first1 == last1)
176  {
177  for (; first2 != last2; ++first2)
178  fpol_het(w, mutations[*first2]);
179  return w;
180  }
181  else if (first2 == last2)
182  {
183  for (; first1 != last1; ++first1)
184  fpol_het(w, mutations[*first1]);
185  return w;
186  }
187  for (; first1 != last1; ++first1)
188  {
189  for (;
190  first2 != last2 && *first1 != *first2
191  && !(mutations[*first2].pos > mutations[*first1].pos);
192  ++first2)
193  // All mutations in this range are Aa
194  {
195  assert(mutations[*first2].pos
196  < mutations[*first1].pos);
197  fpol_het(w, mutations[*first2]);
198  }
199  if (first2 < last2 && *first1 == *first2) // mutation with
200  // index first1
201  // is homozygous
202  {
203  assert(mutations[*first2].pos
204  == mutations[*first1].pos);
205  fpol_hom(w, mutations[*first1]);
206  ++first2; // increment so that we don't re-process
207  // this site as a het next time 'round
208  }
209  else // mutation first1 is heterozygous
210  {
211  fpol_het(w, mutations[*first1]);
212  }
213  }
214  for (; first2 != last2; ++first2)
215  {
216  fpol_het(w, mutations[*first2]);
217  }
218  return w;
219  }
