101 #ifndef NDEBUG
102  if (mcounts.size() != mutations.size())
103  {
104  throw std::runtime_error(
105  "FWDPP DEBUG: mutation container size must equal "
106  "mutation count container size");

115  /*
116  The mutation and haploid_genome containers contain both extinct and extant
117  objects.
118  The former are useful b/c the represent already-allocated memory.
119  The library
120  uses these extinct objects to 'recycle' them into new objects. The
121  function calls
122  below create FIFO queues of where extinct objects are. These queues
123  are passed to
124  mutation and recombination functions and used to decide if recyling
125  is possible or
126  if a new object needs to be 'emplace-back'-ed into a container.
127 
128  The type of the FIFO queue is abstracted with the name
129  fwdpp::fwdpp_internal::recycling_bin_t,
130  which is a C++11 template alias.
131 
132  The details of recycling are implemented in
133  fwdpp/internal/recycling.hpp
134  */
135  auto mut_recycling_bin = make_mut_queue(mcounts);
136  auto gam_recycling_bin = make_haploid_genome_queue(haploid_genomes);
137 
138  // Calculate fitness for each diploid:
139 
140  // create a vector to store fitnesses:
141  std::vector<double> fitnesses(diploids.size());
142  double wbar = 0.; // pop'n mean fitness
143  for (uint_t i = 0; i < N_curr; ++i)
144  {
145  /*
146  Set the count of each haploid_genome to 0.
147 
148  Yes, in the case of > 1 diploid having the exact same haploid_genome
149  index,
150  this is done multiple times.
151  */
152  haploid_genomes[diploids[i].first].n
153  = haploid_genomes[diploids[i].second].n = 0;
154  /*
155  Assign fitness to the i-th individual.
156 
157  ff is a "fitness function", which returns a double. For
158  examples, see
159  fwdpp::multiplicative_diploid, which is a "standard" type of
160  fitness function
161  used in population genetics. "Standard" types of models are
162  defined in
163  fwdpp/fitness_models.hpp.
164  */
165  fitnesses[i] = ff(diploids[i], haploid_genomes, mutations);
166  wbar += fitnesses[i];
167  }
168  wbar /= double(diploids.size());
169 #ifndef NDEBUG
170  for (const auto &g : haploid_genomes)
171  {
172  if (g.n > 0)
173  {
174  throw std::runtime_error(
175  "FWDPP DEBUG: not all haploid_genome counts equal "
176  "zero");
177  }
178  }
179 #endif
180 
181  /*
182  This is a lookup table for rapid sampling of diploids proportional to
183  their fitnesses.
184  This is a unique_ptr wrapper around an object from the GNU Scientific
185  Library. A custom deleter
186  is required to make this work, which is why there is no cleanup call
187  down below.
188  */
189  gsl_ran_discrete_t_ptr lookup(
190  gsl_ran_discrete_preproc(N_curr, fitnesses.data()));
191  const auto parents(diploids); // Copy the parents, which is trivally
192  // fast for the vast majority of use
193  // cases.
194 
195  // Change the population size
196  if (diploids.size() != N_next)
197  {
198  diploids.resize(N_next);
199  }
200 
201  // Fill in the next generation!
202  for (auto &dip : diploids)
203  {
204  // Choose parent 1 based on fitness
205  auto p1 = gsl_ran_discrete(r, lookup.get());
206  // If inbred (w/probability f2), parent2 = parent1, else choose
207  // again based on fitness
208  auto p2 = (f == 1. || (f > 0. && gsl_rng_uniform(r) < f))
209  ? p1
210  : gsl_ran_discrete(r, lookup.get());
211  /*
212  These are the haploid_genomes from each parent.
213  This is a trivial assignment if keys.
214  */
215  auto p1g1 = parents[p1].first;
216  auto p1g2 = parents[p1].second;
217  auto p2g1 = parents[p2].first;
218  auto p2g2 = parents[p2].second;
219 
220  /*
221  The offspring will inherit some manipulation of p1g1 and
222  p1g2.
223  The next two lines do "Mendel".
224  */
225  if (gsl_rng_uniform(r) < 0.5)
226  std::swap(p1g1, p1g2);
227  if (gsl_rng_uniform(r) < 0.5)
228  std::swap(p2g1, p2g2);
229 
230  mutate_recombine_update(
231  r, haploid_genomes, mutations,
232  std::make_tuple(p1g1, p1g2, p2g1, p2g2), rec_pol, mmodel,
233  mu, gam_recycling_bin, mut_recycling_bin, dip, neutral,
234  selected);
235  }
236 #ifndef NDEBUG
237  for (const auto &dip : diploids)
238  {
239  assert(haploid_genomes[dip.first].n > 0);
240  assert(haploid_genomes[dip.first].n <= 2 * N_next);
241  assert(haploid_genomes[dip.second].n > 0);
242  assert(haploid_genomes[dip.second].n <= 2 * N_next);
243  }
244 #endif
245  /*
246  At the end of the above loop, we have a bunch of new diploids
247  that are all recombined and mutated sampling of the parental
248  generation.
249 
250  Our problem is that we no longer know how many times each mutation is
251  present, which
252  is corrected by the following call.
253 
254  Although the implementation of process_haploid_genomes is super-trivial, it
255  is actually the
256  most computationally-expensive part of a simulation once mutation
257  rates are large.
258 
259  Further, the function is hard to optimize. Recall that haploid_genomes store
260  mutations in order
261  according to position. Thus, when we go from a haploid_genome to a position
262  in mcounts, we are
263  accessing the latter container out of order with respect to location
264  in memory. process_haploid_genomes
265  is thus the "scatter" part of a "scatter-gather" idiom. Modern x86
266  CPU have little available
267  for vectorizing such cases. I've experimented with CPU intrinsics to
268  attempt memory prefetches,
269  but never saw any performance improvement, and the code got complex,
270  and possibly less portable.
271 
272  The implementation is in fwdpp/internal/sample_diploid_helpers.hpp
273  */
274  fwdpp_internal::process_haploid_genomes(haploid_genomes, mutations,
275  mcounts);
276 #ifndef NDEBUG
277  for (const auto &mc : mcounts)
278  {
279  if (mc > 2 * N_next)
280  {
281  throw std::runtime_error("mutation size too large");
282  }
283  }
284 #endif

286  /*
287  The last thing to do is handle fixations. In many contexts, we
288  neither want nor need
289  to keep indexes to fixed variants in our haploid_genomes. Such decisions are
290  implemented via
291  simple policies, which are in the variable 'mp'.
292 
293  The implementation is in fwdpp/internal/haploid_genome_cleaner.hpp.
294 
295  The implementation is the "erase/remove idiom" (Effective STL, Item
296  32), but with a twist
297  that the function will exit early if there are no fixations present
298  in the population at
299  the moment.
300 
301  Example policies are fwdpp::remove_nothing and fwdpp::remove_neutral,
302  both found
303  in fwdpp/fwd_functional.hpp. If mp is std::true_type, then all
304  fixations (e.g., neutral
305  and selected) will be removed from all haploid_genomes.
306  */
307  fwdpp_internal::haploid_genome_cleaner(haploid_genomes, mutations,
308  mcounts, 2 * N_next, mp);
309  return wbar;
